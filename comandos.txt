    #docker service scale docker-swarm_backend=1

**Parar y Eliminar Contenedores
    #docker ps
    #docker stop <nombre_o_id_del_contenedor>
    #docker rm <nombre_o_id_del_contenedor>

**Parar y Eliminar Servicios en Docker Swarm
    **Listar los Servicios del Stack
        #docker stack services docker-swarm
    **Parar (escalar a 0) y Eliminar los Servicios
        **Backend
            #docker service scale docker-swarm_backend=0
            #docker service rm docker-swarm_backend
        **Frontend
            #docker service scale docker-swarm_frontend=0
            #docker service rm docker-swarm_frontend
        **DB
            #docker service scale docker-swarm_db=0
            #docker service rm docker-swarm_db
